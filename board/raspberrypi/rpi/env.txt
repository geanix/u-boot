loadaddr=0x1000000
kernel_addr=0x4000000
fdt_addr=0x03000000

load_kernel=ext4load mmc 0:2 $kernel_addr /boot/$image; unzip $kernel_addr $loadaddr
load_fdt=ext4load mmc 0:2 $fdt_addr /boot/$fdtfile
bootcmd=run load_fdt load_kernel; booti $loadaddr - $fdt_addr
