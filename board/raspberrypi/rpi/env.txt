image=Image.gz
loadaddr=0x1000000
kernel_addr=0x4000000
fdt_addr=0x03000000
fdt_high=0xffffffff
bootargs=console=ttyS1,115200n8 root=/dev/mmcblk0p2 rootwait rw quiet
load_kernel=ext4load mmc 0:2 $kernel_addr /boot/$image; unzip $kernel_addr $loadaddr
load_fdt=ext4load mmc 0:2 $fdt_addr /boot/$fdtfile
load_overlays=fdt addr $fdt_addr; for f in $fdt_overlays; do ext4load mmc 0:2 $loadaddr /boot/$f; fdt resize; fdt apply $loadaddr; done
bootcmd=run load_fdt load_overlays load_kernel; booti $loadaddr - $fdt_addr
